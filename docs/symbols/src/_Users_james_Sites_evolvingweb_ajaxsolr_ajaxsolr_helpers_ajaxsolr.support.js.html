  3 /**
  4  * Optionally URL-encodes a string.
  5  *
  6  * @param {Boolean} skip Whether to skip encoding.
  7  * @returns {String} The possibly-encoded string.
  8  */
  9 String.prototype.urlencode = function(skip) {
 10   if (skip) {
 11     return this;
 12   }
 13   else {
 14     return encodeURIComponent(this);
 15   }
 16 }

 18 /**
 19  * Strip whitespace from the beginning and end of a string.
 20  *
 21  * @returns {String} The trimmed string.
 22  */
 23 String.prototype.trim = function() {
 24   return this.replace(/^ +/, '').replace(/ +$/, '');
 25 }

 27 /**
 28  * Returns a date in ISO8601 format.
 29  *
 30  * @returns {Date} The date in ISO8601 format.
 31  */
 32 String.prototype.toDate = function() {
 33   return new Date().setISO8601(this);
 34 }

 58 /**
 59  * @see http://dansnetwork.com/2008/11/01/javascript-iso8601rfc3339-date-parser/ 
 60  */
 61 Date.prototype.setISO8601 = function(dString) {
 62   var regexp = /(\d\d\d\d)(-)?(\d\d)(-)?(\d\d)(T)?(\d\d)(:)?(\d\d)(:)?(\d\d)(\.\d+)?(Z|([+-])(\d\d)(:)?(\d\d))/;
 63 
 64   if (dString.toString().match(new RegExp(regexp))) {
 65     var d = dString.match(new RegExp(regexp));
 66     var offset = 0;
 67 
 68     this.setUTCDate(1);
 69     this.setUTCFullYear(parseInt(d[1], 10));
 70     this.setUTCMonth(parseInt(d[3], 10) - 1);
 71     this.setUTCDate(parseInt(d[5], 10));
 72     this.setUTCHours(parseInt(d[7], 10));
 73     this.setUTCMinutes(parseInt(d[9], 10));
 74     this.setUTCSeconds(parseInt(d[11], 10));
 75     if (d[12]) {
 76       this.setUTCMilliseconds(parseFloat(d[12]) * 1000);
 77     }
 78     else {
 79       this.setUTCMilliseconds(0);
 80     }
 81     if (d[13] != 'Z') {
 82       offset = (d[15] * 60) + parseInt(d[17], 10);
 83       offset *= ((d[14] == '-') ? -1 : 1);
 84       this.setTime(this.getTime() - offset * 60 * 1000);
 85     }
 86   }
 87   else {
 88     this.setTime(Date.parse(dString));
 89   }
 90   return this;
 91 }

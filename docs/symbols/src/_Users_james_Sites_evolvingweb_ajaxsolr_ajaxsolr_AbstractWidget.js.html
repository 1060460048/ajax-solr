  3 /**
  4  * Baseclass for all widgets. 
  5  * 
  6  * Handles the selection of items (filters) and provides abstract hooks for
  7  * child classes.
  8  *
  9  * @param properties A map of fields to set. May be new or public fields.
 10  * @class AbstractWidget
 11  */
 12 AjaxSolr.AbstractWidget = AjaxSolr.Class.extend(
 13   /** @lends AjaxSolr.AbstractWidget.prototype */
 14   {
 15   /** 
 16    * A unique identifier of this widget.
 17    *
 18    * @field 
 19    * @public
 20    * @type String
 21    */
 22   id: null,
 23 
 24   /** 
 25    * A CSS selector representing the "target div" inside the html page.
 26    * All UI changes will be performed inside this empty div.
 27    * 
 28    * @field 
 29    * @public
 30    * @type String
 31    */
 32   target: null,
 33 
 34   /** 
 35    * A CSS selector representing the "container div" inside the html page.
 36    * A widget need not necessarily have a container div.
 37    * 
 38    * @field 
 39    * @public
 40    * @type String
 41    */
 42   container: null,
 43 
 44   /**
 45    * A flag that indicates whether new selected items should replace old ones.
 46    *
 47    * @field
 48    * @private
 49    * @type Boolean
 50    * @default false
 51    */
 52   replace: false,
 53 
 54   /** 
 55    * A flag that indicates whether we should animate the update of the target.
 56    * 
 57    * @field
 58    * @private
 59    * @type Boolean
 60    * @default false
 61    */
 62   animate: false,
 63 
 64   /**
 65    * The list of selected items.
 66    *
 67    * @field
 68    * @private
 69    * @type Array
 70    * @default []
 71    */
 72   selectedItems: [],
 73 
 74   /**
 75    * Add the given items to the list of selected items.
 76    *
 77    * @param {Array} items The items to add.
 78    * @returns {Boolean} Whether the selection changed.
 79    */
 80   selectItems: function(items) {
 81     if (this.replace) {
 82       this.deselectAll();
 83     }
 84     var start = this.selectedItems.length;
 85     for (var i = 0; i < items.length; i++) {
 86       // can't do loop as in deselectItems() because here we are testing if an
 87       // item is not in the array, rather than if an item is in the array
 88       if (!AjaxSolr.contains(this.selectedItems, items[i])) {
 89         this.selectedItems.push(items[i]);
 90       }
 91     }
 92     if (this.selectedItems.length > start) {
 93       this.afterChangeSelection();
 94     }
 95     return this.selectedItems.length > start;
 96   },
 97 
 98   /**
 99    * Removes the given items from the list of selected items.
100    *
101    * @param {Array} items The items to remove.
102    * @returns {Boolean} Whether the selection changed.
103    */
104   deselectItems: function(items) {
105     var start = this.selectedItems.length;
106     for (var i = 0; i < items.length; i++) {
107       for (var j = this.selectedItems.length - 1; j >= 0; j--) {
108         if (this.selectedItems[j] == items[i]) {
109           this.selectedItems.splice(j, 1);
110         }
111       }
112     }
113     if (this.selectedItems.length < start) {
114       this.afterChangeSelection();
115     }
116     return this.selectedItems.length < start;
117   },
118 
119   /**
120    * Removes all items from the current selection.
121    */
122   deselectAll: function() {
123     this.selectedItems = [];
124     this.afterChangeSelection();
125   },
126 
127   // Methods to be overridden by widgets:
128 
129   /**
130    * An abstract hook for child implementations.
131    * Alters the query before it is run.
132    *
133    * @param queryObj The query object built by buildQuery.
134    */
135   alterQuery: function(queryObj) {},
136 
137   /**
138    * An abstract hook for child implementations.
139    * Displays the query before it is run.
140    *
141    * @param queryObj The query object built by buildQuery.
142    */
143   displayQuery: function(queryObj) {},
144 
145   /**
146    * An abstract hook for child implementations.
147    * This method is executed after the Solr response data arrives.
148    *
149    * @param data The Solr response inside a JavaScript object.
150    */
151   handleResult: function(data) {},
152 
153   /** 
154    * An abstract hook for child implementations.
155    * This method need only be defined if animate=true.
156    */
157   startAnimation: function() {},
158 
159   /** 
160    * An abstract hook for child implementations.
161    * This method need only be defined if animate=true.
162    */
163   endAnimation: function() {},
164 
165   /**
166    * An abstract hook for child implementations.
167    * This method should do any necessary one-time initializations.
168    */
169   afterAdditionToManager: function() {},
170 
171   /**
172    * An abstract hook for child implementations.
173    * This method is executed after items are selected or deselected.
174    */
175   afterChangeSelection: function() {}
176 });

 41 /**
 42  * Baseclass for all date facet widgets.
 43  *
 44  * @class AbstractDateFacetWidget
 45  * @augments AjaxSolr.AbstractFacetWidget
 46  */
 47 AjaxSolr.AbstractDateFacetWidget = AjaxSolr.AbstractFacetWidget.extend(
 48   /** @lends AjaxSolr.AbstractDateFacetWidget.prototype */
 49   {
 50   replace: true,
 51 
 52   /**
 53    * Earliest date selectable.
 54    *
 55    * @field
 56    * @public
 57    * @type String
 58    */
 59   minDate: null,
 60 
 61   /**
 62    * Latest date selectable.
 63    *
 64    * @field
 65    * @public
 66    * @type String
 67    */
 68   maxDate: null,
 69 
 70   /**
 71    * Allow users to drill down to this level of granularity.
 72    *
 73    * @field
 74    * @public
 75    * @type String
 76    * @default "MINUTE"
 77    */
 78   granularity: 'MINUTE',
 79 
 80   /**
 81    * The list of date parts, used to determine the next or previous date part.
 82    *
 83    * @static
 84    * @private
 85    */
 86   parts: [
 87     'YEAR',
 88     'MONTH',
 89     'DAY',
 90     'HOUR',
 91     'MINUTE',
 92     'SECOND',
 93     'MILLISECOND'
 94   ],
 95 
 96   /**
 97    * Stop-index for each date part, when appearing in a date in ISO 8601 format.
 98    *
 99    * @static
100    * @private
101    */
102   indices: {
103     'YEAR'   : 4,
104     'MONTH'  : 7,
105     'DAY'    : 10,
106     'HOUR'   : 13,
107     'MINUTE' : 16,
108     'SECOND' : 19
109   },
110 
111   alterQuery: function(queryObj) {
112     if (this.selectedItems.length) {
113       var gap = this.selectedItems[0][1].split('+1')[1];
114 
115       queryObj.dates.push({
116         field: this.fieldName,
117         start: this.selectedItems[0][0] + '/' + gap,
118         end:   this.selectedItems[0][1] + '/' + gap,
119         gap:   '+1' + this.parts[this.indexOfPart(gap) + 1]
120       });
121     }
122     else {
123       queryObj.dates.push({
124         field: this.fieldName,
125         start: this.minDate + '/YEAR',
126         end:   this.maxDate + '+1YEAR/YEAR',
127         gap:   '+1YEAR'
128       });
129     }
130     queryObj.fq = queryObj.fq.concat(this.getItems());
131   },
132 
133   unclickHandler: function(value, gap) {
134     var me = this;
135 
136     // if called from outside this class, gap will not be set
137     if (gap === undefined) {
138       gap = value[1].split('+1')[1];
139     }
140 
141     // deselecting a date selects the previous level of granularity
142     gap = this.parts[this.indexOfPart(gap) - 1];
143 
144     return function() {
145       if (gap) {
146         me.manager.selectItems(me.id, [ me.getValue(value[0], gap) ]);
147       }
148       else {
149         me.manager.deselectWidget(me.id);
150       }
151       return false;
152     }
153   },
154 
155   unclickText: function(value) {
156     return value[0].toDate().toLongDateString(value[1].split('+1')[1]);
157   },
158 
159   clickText: function(value) {
160     return value[0].toDate().datePartString(this.facetDates.gap.substring(2));
161   },
162 
163   /**
164    * Given a date and a granularity, returns a data range in which the dates
165    * are in ISO8601 format truncated to the given level of granularity.
166    *
167    * @param {String} date The date.
168    * @param {String} granularity The granularity.
169    * @returns {String[]} The date range.
170    */
171   getValue: function(date, granularity) {
172     index = this.indices[granularity];
173     date = date.substring(0, index) + '0000-01-01T00:00:00Z'.substring(index);
174     return [ date, date + '+1' + granularity ];
175   },
176 
177   /**
178    * Array.indexOf() is undefined in IE, but rather than define it globally,
179    * I just write this one function for the one case in which I need it.
180    *
181    * @param {String} part The date part to search for.
182    * @returns {Number} If found, the index of the date part; -1, otherwise.
183    */
184   indexOfPart: function(part) {
185     for (var i = 0; i < this.parts.length; i++) {
186       if(this.parts[i] == part) {
187         return i;
188       }
189     }
190     return -1;
191   }
192 });

  3 /**
  4  * Baseclass for all result widgets.
  5  *
  6  * @class AbstractResultWidget
  7  * @augments AjaxSolr.AbstractWidget
  8  */
  9 AjaxSolr.AbstractResultWidget = AjaxSolr.AbstractWidget.extend(
 10   /** @lends AjaxSolr.AbstractResultWidget.prototype */
 11   {
 12   /**
 13    * The set of fields to return.
 14    *
 15    * @field
 16    * @public
 17    * @default []
 18    */
 19   fields: [],
 20 
 21   /**
 22    * The number of fields to return.
 23    *
 24    * @field
 25    * @public
 26    * @default 0
 27    */
 28   rows: 0,
 29 
 30   alterQuery: function(queryObj) {
 31     queryObj.fl = queryObj.fl.concat(this.fields);
 32     if (this.rows > queryObj.rows) {
 33       queryObj.rows = this.rows;
 34     }
 35   }
 36 });

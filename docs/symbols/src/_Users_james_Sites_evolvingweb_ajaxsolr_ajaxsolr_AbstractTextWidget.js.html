  3 /**
  4  * Baseclass for all free text widgets.
  5  *
  6  * @class AbstractTextWidget
  7  * @augments AjaxSolr.AbstractWidget
  8  */
  9 AjaxSolr.AbstractTextWidget = AjaxSolr.AbstractWidget.extend(
 10   /** @lends AjaxSolr.AbstractTextWidget.prototype */
 11   {
 12   replace: true,
 13 
 14   alterQuery: function(queryObj) {
 15     queryObj.q = queryObj.q.concat(this.getItems());
 16   },
 17 
 18   /**
 19    * Returns all the selected items as query items.
 20    *
 21    * @returns {QueryItem[]}
 22    */
 23   getItems: function() {
 24     var items = [];
 25     for (var i = 0; i < this.selectedItems.length; i++) {
 26       items.push(new AjaxSolr.QueryItem({
 27         value: this.selectedItems[i]
 28       }));
 29     }
 30     return items;
 31   }
 32 });

  3 /**
  4  * Represents a Solr query.
  5  *
  6  * @param properties A map of fields to set. Refer to the list of public fields.
  7  * @class QueryItem
  8  */
  9 AjaxSolr.QueryItem = AjaxSolr.Class.extend(
 10   /** @lends AjaxSolr.QueryItem.prototype */
 11   {
 12   /**
 13    * The value.
 14    *
 15    * @field
 16    * @public
 17    * @type String
 18    */
 19   value: null,
 20 
 21   /**
 22    * Transforms this item into Solr syntax.
 23    *
 24    * @param {Boolean} skip Whether to skip encoding the value.
 25    * @returns {String} Solr Query syntax.
 26    */
 27   toSolr: function(skip) {
 28     return this.value.urlencode(skip);
 29   },
 30 
 31   /**
 32    * Prepares this item for inclusion in the URL hash.
 33    *
 34    * @returns {String} A key-value pair for the URL hash.
 35    */
 36   toHash: function() {
 37     return this.value.urlencode();
 38   },
 39 
 40   /**
 41    * Parses a key-value pair from the URL hash.
 42    *
 43    * @param {String} string A key-value pair from the URL hash.
 44    */
 45   parseHash: function(string) {
 46     this.value = decodeURIComponent(string);
 47   }
 48 });

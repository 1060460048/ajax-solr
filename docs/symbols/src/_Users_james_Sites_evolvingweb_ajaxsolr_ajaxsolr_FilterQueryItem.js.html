  3 /**
  4  * Represents a Solr filter query.
  5  *
  6  * @param properties A map of fields to set. Refer to the list of public fields.
  7  * @class FilterQueryItem
  8  */
  9 AjaxSolr.FilterQueryItem = AjaxSolr.Class.extend(
 10   /** @lends AjaxSolr.FilterQueryItem.prototype */
 11   {
 12   /**
 13    * The field name.
 14    *
 15    * @field
 16    * @public
 17    * @type String
 18    */
 19   field: null,
 20 
 21   /**
 22    * The value.
 23    *
 24    * @field
 25    * @public
 26    * @type String
 27    */
 28   value: null,
 29 
 30   /**
 31    * Whether the value is publicly viewable.
 32    *
 33    * @field
 34    * @public
 35    * @type Boolean
 36    */
 37   hidden: false,
 38 
 39   /**
 40    * The widget that manages this item.
 41    *
 42    * @field
 43    * @public
 44    * @type String
 45    */
 46   widgetId: null,
 47 
 48   /**
 49    * Transforms this item into Solr syntax.
 50    *
 51    * @param {Boolean} skip Whether to skip encoding the value.
 52    * @returns {String} Solr Filter Query syntax.
 53    */
 54   toSolr: function(skip) {
 55     return this.field + ':' + this.getValue().urlencode(skip);
 56   },
 57 
 58   /**
 59    * Prepares this item for inclusion in the URL hash.
 60    *
 61    * @returns {String} A key-value pair for the URL hash.
 62    */
 63   toHash: function() {
 64     return this.widgetId + ':' + this.getValue().urlencode();
 65   },
 66 
 67   /**
 68    * Parses a key-value pair from the URL hash.
 69    *
 70    * @param {String} string A key-value pair from the URL hash.
 71    */
 72   parseHash: function(string) {
 73     var parts = string.split(':');
 74     this.widgetId = parts[0];
 75     this.value = decodeURIComponent(parts.slice(1, parts.length).join(':'));
 76     this.value = this.value.substring(1, this.value.length - 1);
 77 
 78     // handle multiple values
 79     var values = this.value.split(' TO ');
 80     if (values.length == 2) {
 81       this.value = values;
 82     }
 83   },
 84 
 85   /**
 86    * Flattens the value into a quoted string.
 87    *
 88    * @returns {String} A quoted string.
 89    */
 90   getValue: function() {
 91     if (AjaxSolr.isArray(this.value) && this.value.length == 2) {
 92       return '[' + this.value[0] + ' TO ' + this.value[1] + ']';
 93     } else {
 94       return '"' + this.value + '"';
 95     }
 96   }
 97 });

 13 /**
 14  * A class 'extends' itself into a subclass.
 15  *
 16  * @static
 17  * @param properties The properties of the subclass.
 18  * @returns A function that represents the subclass.
 19  */
 20 AjaxSolr.Class.extend = function (properties) {
 21   var klass = this; // Safari dislikes 'class'
 22   // The subclass is just a function that when called, instantiates itself.
 23   // Nothing is _actually_ shared between _instances_ of the same class.
 24   var subClass = function (options) {
 25     // 'this' refers to the subclass, which starts life as an empty object.
 26     // Add its parent's properties, its own properties, and any passed options.
 27     AjaxSolr.extend(this, new klass(options), properties, options);
 28   }
 29   // Allow the subclass to extend itself into further subclasses.
 30   subClass.extend = this.extend;
 31   return subClass;
 32 };

 34 /**
 35  * A simplified version of jQuery's extend function.
 36  *
 37  * @static
 38  * @see http://ajax.googleapis.com/ajax/libs/jquery/1.2.6/jquery.js
 39  */
 40 AjaxSolr.extend = function () {
 41   var target = arguments[0] || {}, i = 1, length = arguments.length, options;
 42   for (; i < length; i++) {
 43     if ((options = arguments[i]) != null) {
 44       for (var name in options) {
 45         var src = target[name], copy = options[name];
 46         if (target === copy) {
 47           continue;
 48         }
 49         if (copy && typeof copy == 'object' && !copy.nodeType) {
 50           target[name] = AjaxSolr.extend(src || (copy.length != null ? [] : {}), copy);
 51         }
 52         else if (copy && src && typeof copy == 'function' && typeof src == 'function') {
 53           target[name] = (function(superfn, fn) {
 54             return function () {
 55               var tmp = this._super, ret;
 56               this._super = superfn;
 57               ret = fn.apply(this, arguments);
 58               this._super = tmp;
 59               return ret;
 60             };
 61           })(src, copy);
 62         }
 63         else if (copy !== undefined) {
 64           target[name] = copy;
 65         }
 66       }
 67     }
 68   }
 69   return target;
 70 };

 72 /**
 73  * @static
 74  * @param value A value.
 75  * @param array An array.
 76  * @returns {Boolean} Whether value exists in the array.
 77  */
 78 AjaxSolr.inArray = function (value, array) {
 79   if (array) {
 80     for (var i = 0, l = array.length; i < l; i++) {
 81       if (AjaxSolr.equals(array[i], value)) {
 82         return i;
 83       }
 84     }
 85   }
 86   return -1;
 87 };

 89 /**
 90  * @static
 91  * @param foo A value.
 92  * @param bar A value.
 93  * @returns {Boolean} Whether the two given values are equal.
 94  */
 95 AjaxSolr.equals = function (foo, bar) {
 96   if (AjaxSolr.isArray(foo) && AjaxSolr.isArray(bar)) {
 97     if (foo.length !== bar.length) {
 98       return false;
 99     }
100     for (var i = 0, l = foo.length; i < l; i++) {
101       if (foo[i] !== bar[i]) {
102         return false;
103       }
104     }
105     return true;
106   }
107   else if (AjaxSolr.isRegExp(foo) && AjaxSolr.isString(bar)) {
108     return bar.match(foo);
109   }
110   else if (AjaxSolr.isRegExp(bar) && AjaxSolr.isString(foo)) {
111     return foo.match(bar);
112   }
113   else {
114     return foo === bar;
115   }
116 };

118 /**
119  * Can't use toString.call(obj) === "[object Array]", as it may return
120  * "[xpconnect wrapped native prototype]", which is undesirable.
121  *
122  * @static
123  * @see http://thinkweb2.com/projects/prototype/instanceof-considered-harmful-or-how-to-write-a-robust-isarray/
124  * @see http://ajax.googleapis.com/ajax/libs/prototype/1.6.0.3/prototype.js
125  */
126 AjaxSolr.isArray = function (obj) {
127   return obj != null && typeof obj == 'object' && 'splice' in obj && 'join' in obj;
128 };
129 
130 /**
131  * @param obj Any object.
132  * @returns {Boolean} Whether the object is a RegExp object.
133  */
134 AjaxSolr.isRegExp = function (obj) {
135   return obj != null && (typeof obj == 'object' || typeof obj == 'function') && 'ignoreCase' in obj;
136 };
137 
138 /**
139  * @param obj Any object.
140  * @returns {Boolean} Whether the object is a String object.
141  */
142 AjaxSolr.isString = function (obj) {
143   return obj != null && typeof obj == 'string';
144 };

 13 /**
 14  * A class 'extends' itself into a subclass.
 15  *
 16  * @static
 17  * @param properties The properties of the subclass.
 18  * @returns A function that represents the subclass.
 19  */
 20 AjaxSolr.Class.extend = function(properties) {
 21   var klass = this; // Safari dislikes 'class'
 22   // The subclass is just a function that when called, instantiates itself.
 23   // Nothing is _actually_ shared between _instances_ of the same class.
 24   var subClass = function(options) {
 25     // 'this' refers to the subclass, which starts life as an empty object.
 26     // Add its parent's properties, its own properties, and any passed options.
 27     AjaxSolr.extend(this, new klass(options), properties, options);
 28   }
 29   // Allow the subclass to extend itself into further subclasses.
 30   subClass.extend = this.extend;
 31   return subClass;
 32 }

 34 /**
 35  * Returns whether an item is in the given array or not.
 36  * Don't add this to Array.prototype as it may break for loops.
 37  *
 38  * @static
 39  * @param {Array} items The array.
 40  * @param item An item that may or may not be in the array.
 41  * @returns {Boolean} Whether the item is in the given array.
 42  * @see http://www.prototypejs.org/api/array
 43  */
 44 AjaxSolr.contains = function(items, item) {
 45   for (var i = 0; i < items.length; i++) {
 46     if (items[i] == item) {
 47       return true;
 48     }
 49   }
 50   return false;
 51 }

 55 /**
 56  * A copy of Drupal's Drupal.theme function.
 57  *
 58  * @static
 59  * @see http://cvs.drupal.org/viewvc.py/drupal/drupal/misc/drupal.js?revision=1.58
 60  */
 61 AjaxSolr.theme = function(func) {
 62   for (var i = 1, args = []; i < arguments.length; i++) {
 63     args.push(arguments[i]);
 64   }
 65 
 66   return (AjaxSolr.theme[func] || AjaxSolr.theme.prototype[func]).apply(this, args);
 67 };

 69 /**
 70  * Can't use toString.call(obj) === "[object Array]", as it will may return
 71  * "[xpconnect wrapped native prototype]", which is undesirable.
 72  *
 73  * @static
 74  * @see http://thinkweb2.com/projects/prototype/instanceof-considered-harmful-or-how-to-write-a-robust-isarray/
 75  * @see http://ajax.googleapis.com/ajax/libs/prototype/1.6.0.3/prototype.js
 76  */
 77 AjaxSolr.isArray = function(obj) {
 78   return obj != null && typeof obj == 'object' && 'splice' in obj && 'join' in obj;
 79 }

 81 /**
 82  * A simplified version of jQuery's extend function.
 83  *
 84  * @static
 85  * @see http://ajax.googleapis.com/ajax/libs/jquery/1.2.6/jquery.js
 86  */
 87 AjaxSolr.extend = function() {
 88   var target = arguments[0] || {}, i = 1, length = arguments.length, options;
 89   for (; i < length; i++) {
 90     if ((options = arguments[i]) != null) {
 91       for (var name in options) {
 92         var src = target[name], copy = options[name];
 93         if (target === copy) {
 94           continue;
 95         }
 96         if (copy && typeof copy == 'object' && !copy.nodeType) {
 97           target[name] = AjaxSolr.extend(src || (copy.length != null ? [] : {}), copy);
 98         }
 99         else if (copy !== undefined) {
100           target[name] = copy;
101         }
102       }
103     }
104   }
105   return target;
106 }

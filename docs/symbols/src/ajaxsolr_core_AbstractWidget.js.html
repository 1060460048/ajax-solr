1 /**
  2  * Baseclass for all widgets. 
  3  * 
  4  * Provides abstract hooks for child classes.
  5  *
  6  * @param properties A map of fields to set. May be new or public fields.
  7  * @class AbstractWidget
  8  */
  9 AjaxSolr.AbstractWidget = AjaxSolr.Class.extend(
 10   /** @lends AjaxSolr.AbstractWidget.prototype */
 11   {
 12   /** 
 13    * A unique identifier of this widget.
 14    *
 15    * @field 
 16    * @public
 17    * @type String
 18    */
 19   id: null,
 20 
 21   /** 
 22    * The CSS selector for this widget's target HTML element, e.g. a specific
 23    * <tt>div</tt> or <tt>ul</tt>. A Widget is usually implemented to perform
 24    * all its UI changes relative to its target HTML element.
 25    * 
 26    * @field 
 27    * @public
 28    * @type String
 29    */
 30   target: null,
 31 
 32   /**
 33    * A reference to the widget's manager. For internal use only.
 34    *
 35    * @field
 36    * @private
 37    * @type AjaxSolr.AbstractManager
 38    */
 39   manager: null,
 40 
 41   /**
 42    * The offset parameter. Set this field to make the widget reset the offset
 43    * parameter to the given value on each request.
 44    *
 45    * @field
 46    * @public
 47    * @type Number
 48    */
 49   start: undefined,
 50 
 51   /**
 52    * The Solr servlet for this widget. You may prepend the servlet with a core
 53    * if using multiple cores. If none is set, it will default to the manager's
 54    * servlet.
 55    *
 56    * @field
 57    * @public
 58    * @type String
 59    */
 60   servlet: undefined,
 61 
 62   /**
 63    * An abstract hook for child implementations.
 64    *
 65    * <p>This method should do any necessary one-time initializations.</p>
 66    */
 67   init: function () {},
 68 
 69   /** 
 70    * An abstract hook for child implementations.
 71    *
 72    * <p>This method is executed before the Solr request is sent.</p>
 73    */
 74   beforeRequest: function () {},
 75 
 76   /**
 77    * An abstract hook for child implementations.
 78    *
 79    * <p>This method is executed after the Solr response is received.</p>
 80    */
 81   afterRequest: function () {},
 82 
 83   /**
 84    * A proxy to the manager's doRequest method.
 85    *
 86    * @param {Boolean} [start] The Solr start offset parameter.
 87    * @param {String} [servlet] The Solr servlet to send the request to.
 88    */
 89   doRequest: function (start, servlet) {
 90     this.manager.doRequest(start || this.start, servlet || this.servlet);
 91   }
 92 });

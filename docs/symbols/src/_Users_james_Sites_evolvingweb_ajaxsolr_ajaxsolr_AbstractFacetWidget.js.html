  3 /**
  4  * Baseclass for all facet widgets.
  5  *
  6  * @class AbstractFacetWidget
  7  * @augments AjaxSolr.AbstractWidget
  8  */
  9 AjaxSolr.AbstractFacetWidget = AjaxSolr.AbstractWidget.extend(
 10   /** @lends AjaxSolr.AbstractFacetWidget.prototype */
 11   {
 12   /**
 13    * The field to facet on.
 14    *
 15    * @field
 16    * @public
 17    * @type String
 18    */
 19   fieldName: null,
 20 
 21   /**
 22    * This field contains the human-readable versions of the values stored in
 23    * the field fieldName.
 24    *
 25    * @field
 26    * @public
 27    * @type String
 28    */
 29   humanFieldName: null,
 30 
 31   /**
 32    * Maximum number of facet values to display.
 33    *
 34    * @field
 35    * @public
 36    * @type Number
 37    */
 38   limit: null,
 39 
 40   /**
 41    * Facet fields returned by Solr.
 42    *
 43    * @field
 44    * @private
 45    */
 46   facetFields: null,
 47 
 48   /**
 49    * Facet dates returned by Solr.
 50    *
 51    * @field
 52    * @private
 53    */
 54   facetDates: null,
 55 
 56   alterQuery: function(queryObj) {
 57     queryObj.fields.push(this.fieldName);
 58     queryObj.fields.push(this.humanFieldName);
 59     queryObj.fq = queryObj.fq.concat(this.getItems());
 60   },
 61 
 62   handleResult: function(data) {
 63     if (data.facet_counts) {
 64       // we want to display only the human-readable facet values
 65       this.facetFields = data.facet_counts.facet_fields[this.humanFieldName];
 66       this.facetDates = data.facet_counts.facet_dates[this.humanFieldName];
 67       // allow the child implementation to handle the result.
 68       this._handleResult();
 69     }
 70   },
 71 
 72   /**
 73    * An abstract hook for child implementations.
 74    * Allow the child to handle the result without parsing the response.
 75    *
 76    * @throws If not defined in child implementation.
 77    */
 78   _handleResult: function() {
 79     throw 'Abstract method _handleResult';
 80   },
 81 
 82   /**
 83    * Returns all the selected items as filter query items.
 84    *
 85    * @returns {FilterQueryItem[]}
 86    */
 87   getItems: function() {
 88     var items = [];
 89     for (var i = 0; i < this.selectedItems.length; i++) {
 90       items.push(new AjaxSolr.FilterQueryItem({
 91         // the facet value that is selectable is the human-readable value, so
 92         // the filter query item should use the field for human-readable values
 93         field: this.humanFieldName,
 94         value: this.selectedItems[i],
 95         widgetId: this.id
 96       }));
 97     }
 98     return items;
 99   },
100 
101   /**
102    * Returns a function to deselect the given value.
103    *
104    * @param value The given value.
105    * @returns {Function}
106    */
107   unclickHandler: function(value) {
108     var me = this;
109     return function() {
110       me.manager.deselectItems(me.id, [ value ]);
111       return false;
112     }
113   },
114 
115   /**
116    * Returns a function to select the given value.
117    *
118    * @param value The given value.
119    * @returns {Function}
120    */
121   clickHandler: function(value) {
122     var me = this;
123     return function() {
124       me.manager.selectItems(me.id, [ value ]);
125       return false;
126     }
127   },
128 
129   /**
130    * Returns a string representation of the value to deselect.
131    *
132    * @param value The value to deselect.
133    * @returns {String}
134    */
135   unclickText: function(value) {
136     return value;
137   },
138 
139   /**
140    * Returns a string representation of the value to select.
141    *
142    * @param value The value to select.
143    * @returns {String}
144    */
145   clickText: function(value) {
146     return value;
147   }
148 });
